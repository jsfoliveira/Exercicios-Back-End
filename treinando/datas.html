-- datas -- consultar data e hora atuais SELECT CURRENT_DATE(); -- YYYY-MM-DD SELECT NOW(); -- YYYY-MM-DD HH:MM:SS SELECT YEAR(CURRENT_DATE()); -- retorna o ano atual SELECT HOUR(NOW()); -- retorna a hora atual -- DATADIFF: calcular a diferença em dias entre duas datas SELECT DATEDIFF('2020-01-01', '2020-01-31'); -- -30, ou seja, a primeira data é 30 dias antes da segunda -- TIMEDIFF: diferença de tempo entre dois horários SELECT TIMEDIFF('08:30:10', '09:30:10'); -- -01:00:00, ou seja, há 1 hora de diferença entre os horários -- Monte uma query que exiba a diferença de dias entre '2030-01-20' e hoje. SELECT TIMEDIFF('2030-01-20', CURRENT_DATE()); -- Monte uma query exiba a diferença de horas entre '10:25:45' e '11:00:00'. SELECT TIMEDIFF('10:25:45', '11:00:00');
